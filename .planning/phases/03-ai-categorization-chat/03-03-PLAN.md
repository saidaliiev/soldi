// supabase/functions/_shared/chat-schemas.ts (strict everywhere)
export const ChatRequest = z.object({
  message: z.string().min(1).max(500),
  history: z.array(z.object({
    role: z.enum(['user','assistant']),
    text: z.string().max(2000),
  })).max(20).default([]),
}).strict();

export const QueryType = z.enum([
  'sum_by_category',
  'count_by_category',
  'sum_by_month',
  'top_merchants',
  'compare_periods',
  'last_n_transactions_aggregate',  // returns counts/sums only, never row-level
]);

export const ToolInput = z.object({
  query_type: QueryType,
  filters: z.object({
    date_from: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
    date_to: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
    category_slugs: z.array(z.string()).max(10).optional(),
    currency: z.enum(['EUR','UAH']).optional(),
  }).strict(),
}).strict();

export const ChartPayload = z.discriminatedUnion('kind', [
  z.object({ kind: z.literal('sparkline'), values: z.array(z.number()).min(2).max(60), kpi: z.number().optional(), unit: z.literal('EUR').optional() }).strict(),
  z.object({ kind: z.literal('donut'), slices: z.array(z.object({ label: z.string().max(40), value: z.number(), color: z.enum(['accent','sage','accentSoft','textMuted']) }).strict()).min(1).max(8), kpi: z.number().optional(), unit: z.literal('EUR').optional() }).strict(),
  z.object({ kind: z.literal('bar'), bars: z.array(z.object({ label: z.string().max(40), value: z.number() }).strict()).min(1).max(5), kpi: z.number().optional(), unit: z.literal('EUR').optional() }).strict(),
]);

export const ChatResponse = z.object({
  text: z.string().min(1).max(800),    // Sonnet prose; max 3 sentences enforced by system prompt
  chart: ChartPayload.optional(),
}).strict();

// Tool result returned to Sonnet (after sql-runner executes)
export const QueryResult = z.object({
  query_type: QueryType,
  filters_applied: ToolInput.shape.filters,
  rows: z.array(z.record(z.string(), z.union([z.string(), z.number(), z.null()]))).max(50),
  clamped_date_range: z.boolean(),
}).strict();
```

  readonly isAwaitingResponse: boolean;
  readonly lastError: string | null;

The Edge Function is a thin orchestrator: it validates the JWT, parses the request, runs deterministic JS reducers over the FactsPack, and returns Sonnet's final answer.     **leakDetector.ts (defense-in-depth for CHAT-04, client-side):**
    ```ts
    import type { FactsPack } from './factsPackTypes';

    /**
     * Returns true if the assistant prose mentions any of the FactsPack's merchant_key values.
     * merchant_key is already normalized (lowercase, no diacritics, Cyrillic preserved), so the
     * check is a case-insensitive substring match after lowercasing the prose.
     *
     * False-positive risk: short or generic keys (e.g., a 2-char key) could match unrelated words.
     * Mitigation: skip keys shorter than 3 chars (configurable). All real merchants in the synthetic
     * generator are >= 3 chars normalized.
     */
    export function detectMerchantLeak(prose: string, factsPack: FactsPack): boolean {
      if (!prose || !factsPack?.top_merchants_by_month?.length) return false;
      const hay = prose.toLowerCase();
      for (const entry of factsPack.top_merchants_by_month) {
        const key = entry.merchant_key;
        if (!key || key.length < 3) continue;
        if (hay.includes(key)) return true;
      }
      return false;
    }
    ```
